288     /**
289      * Test a signed JWT with a given subject.
290      *
291      * @throws JOSEException Test fails if this exception is thrown
292      * @throws ParseException Test fails if this exception is thrown
293      * @throws IOException Test fails if this exception is thrown
294      */
295     @Test
296     public void getSignedSubjectTest() throws JOSEException, IOException, ParseException {
297         Request httpServletRequest = mock(Request.class);
298 
299         String file = this.getClass().getResource("/rfc7515-a1.jwk").getFile();
300         JWKSet jwkset = JWKSet.load(new File(file));
301         List<JWK> jwks = jwkset.getKeys();
302         OctetSequenceKey key = OctetSequenceKey.class.cast(jwks.iterator().next());
303         SecretKey secret = key.toSecretKey();
304 
305         JWSSigner signer = new MACSigner(secret.getEncoded());
306 
307         JWTClaimsSet claimsSet = new JWTClaimsSet.Builder().subject("alice").issuer("https://example.com")
308                 .expirationTime(new Date(new Date().getTime() + 60 * 1000)).build();
309 
310         SignedJWT signedJWT = new SignedJWT(new JWSHeader(JWSAlgorithm.HS256), claimsSet);
311 
312         signedJWT.sign(signer);
313 
314         final Map<String, String> headers = new HashMap<String, String>();
315         headers.put("Authorization", "Bearer " + signedJWT.serialize());
316 
317         Enumeration<String> headerNames = Collections.enumeration(headers.keySet());
318         Enumeration<String> headerValues = Collections.enumeration(headers.values());
319 
320         when(httpServletRequest.getHeaderNames()).thenReturn(headerNames);
321         when(httpServletRequest.getHeaders("Authorization")).thenReturn(headerValues);
322 
323         doAnswer(new Answer<String>() {
324             @Override
325             public String answer(InvocationOnMock invocation) throws Throwable {
326                 Object[] args = invocation.getArguments();
327                 return headers.get((String) args[0]);
328             }
329         }).when(httpServletRequest).getHeader("Authorization");
330 
331         assertNotNull(httpServletRequest.getHeader("Authorization"));
332 
333         Properties properties = new Properties();
334         properties.setProperty(JwtAuthenticator.JWS_ALGORITHM, "HS256");
335         properties.setProperty(JwtAuthenticator.JWS_JWK_SET_FILE, file);
336         properties.setProperty(JwtAuthenticator.JWT_MAX_CLOCK_SKEW, Integer.toString(Integer.MAX_VALUE));
337 
338         String value = JwtAuthenticator.parseAuthorization(properties, httpServletRequest, "sub");
339 
340         assertEquals("alice", value);
